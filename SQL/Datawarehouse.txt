CREATE TABLE DW_MemberType (
    MemTypeCode CHAR(3) PRIMARY KEY,
    MemTypeName VARCHAR2(20) NOT NULL,
    MemTypeDiscount NUMBER(2),
    EffectiveDate DATE DEFAULT SYSDATE,  -- เพิ่ม EffectiveDate
    EndDate DATE DEFAULT NULL,           -- เพิ่ม EndDate
    SourceSystem VARCHAR2(50)           -- เพิ่ม SourceSystem
);

CREATE TABLE DW_EventType (
    EvTypeCode CHAR(3) PRIMARY KEY,
    EvTypeName VARCHAR2(20) NOT NULL,
    EvTypeDescription VARCHAR2(40),
    EffectiveDate DATE DEFAULT SYSDATE,  -- เพิ่ม EffectiveDate
    EndDate DATE DEFAULT NULL,           -- เพิ่ม EndDate
    SourceSystem VARCHAR2(50)           -- เพิ่ม SourceSystem
);

CREATE TABLE DW_Licensor (
    LicCode CHAR(3) PRIMARY KEY,
    LicName VARCHAR2(20) NOT NULL,
    LicEmail VARCHAR2(30) NOT NULL,
    LicTel VARCHAR2(15) NOT NULL,
    EffectiveDate DATE DEFAULT SYSDATE,  -- เพิ่ม EffectiveDate
    EndDate DATE DEFAULT NULL,           -- เพิ่ม EndDate
    SourceSystem VARCHAR2(50)           -- เพิ่ม SourceSystem
);

CREATE SEQUENCE DW_Member_Seq START WITH 1 INCREMENT BY 1;

CREATE TABLE DW_Member (
    MemSK NUMBER PRIMARY KEY,
    MemCode CHAR(3) NOT NULL,
    MemName VARCHAR2(50),
    MemTel VARCHAR2(15),
    MemSTDate DATE,
    MemEndDate DATE,
    MemTypeCode CHAR(3),
    EffectiveDate DATE DEFAULT SYSDATE, 
    EndDate DATE DEFAULT NULL,
    SourceSystem VARCHAR2(50),  -- เพิ่ม SourceSystem
    CONSTRAINT FK_DW_MemberType FOREIGN KEY (MemTypeCode)
        REFERENCES DW_MemberType(MemTypeCode) ON DELETE SET NULL
);

CREATE OR REPLACE TRIGGER DW_Member_BI
BEFORE INSERT ON DW_Member
FOR EACH ROW
BEGIN
  SELECT DW_Member_Seq.NEXTVAL INTO :NEW.MemSK FROM DUAL;
END;
/
