//Cinema1
CREATE  TABLE  MemberType
	(MemTypeCode		CHAR(3)		PRIMARY  KEY,
	  MemTypeName 		VARCHAR2(20)	NOT  NULL,
	  MemTypeDiscount  	NUMBER(2));

CREATE  TABLE  EventType
	(EvTypeCode		CHAR(3)		PRIMARY  KEY,
	  EvTypeName 		VARCHAR2(20)	NOT  NULL,
	EEvTypeDescription	VARCHAR2(40));

CREATE  TABLE  Licensor
	(LicCode		CHAR(3)		PRIMARY  KEY,
	 LicName 		VARCHAR2(20)	NOT  NULL,
	 LicEmail 		VARCHAR2(30)	NOT  NULL,
	LicTel 		VARCHAR2(15)	NOT  NULL);

CREATE  TABLE  Member
	(MemCode		CHAR(3)		PRIMARY  KEY,
	 MemName 		VARCHAR2(20)	NOT  NULL,
	MemTel		VARCHAR2(15)	NOT  NULL,
	  MemSTDate  	DATE,	
	MemEndDate	DATE,
	MemTypeCode CHAR(3),
	CONSTRAINT	Fk_MemType	FOREIGN  KEY(MemTypeCode)
		REFERENCES		MemberType(MemTypeCode));

CREATE  TABLE  Movie
	(MovieCode		CHAR(3)		PRIMARY  KEY,
	 MovieName 		VARCHAR2(20)	NOT  NULL,
	 MovieTitle 		VARCHAR2(50)	NOT  NULL,
	MovieWriter 		VARCHAR2(20)	NOT  NULL,
	MovieDirector 		VARCHAR2(20)	NOT  NULL,
	MovieRating 		NUMBER(2),
	MovieDuration 		DATE ,
	MovieCost 		NUMBER(4)	NOT NULL,
	LicCode		CHAR(3),
	CONSTRAINT	Fk_Lic	FOREIGN  KEY(LicCode)
		REFERENCES		Licensor(LicCode));

CREATE  TABLE  Ticket
	(TicketCode		NUMBER(4)	PRIMARY  KEY,
	  TicketDate 		DATE,
	  MemCode		CHAR(3),
	CONSTRAINT	Fk_Mem	FOREIGN  KEY(MemCode)
		REFERENCES		Member(MemCode));

CREATE  TABLE  ShowTime
	(SHTCode		CHAR(4)	PRIMARY  KEY,
	  SHTSTTime		DATE,
	  SHTEndTime		DATE,
	SHTLang			VARCHAR2(15),
	SHTCapacity		NUMBER(3),
	MovieCode		CHAR(3),
	CONSTRAINT	Fk_Movie	FOREIGN  KEY(MovieCode)
		REFERENCES		Movie(MovieCode));

CREATE TABLE ShowTime_Ticket
(
  SHTCode     CHAR(4),        TicketCode  NUMBER(4),    
  CONSTRAINT pk_ShowTime_Ticket PRIMARY KEY (SHTCode, TicketCode),   
  CONSTRAINT fk_ShowTime FOREIGN KEY (SHTCode) REFERENCES ShowTime(SHTCode),   
  CONSTRAINT fk_Ticket FOREIGN KEY (TicketCode) REFERENCES Ticket(TicketCode)  
);

CREATE  TABLE  Event
	(EventCode		CHAR(3)		PRIMARY  KEY,
	  EventName 		VARCHAR2(20)	NOT  NULL,
	EventSTDate		DATE,
	EvenEndDate		DATE,
	EvTypeCode		CHAR(3),
	SHTCode			CHAR(4),
	CONSTRAINT	Fk_EvType	FOREIGN  KEY(EvTypeCode)
		REFERENCES		EventType(EvTypeCode),
	CONSTRAINT	Fk_SHT	FOREIGN  KEY(SHTCode)
		REFERENCES		ShowTime(SHTCode));


CREATE TABLE DetailEvent
(
  DetailEventID  NUMBER(10) PRIMARY KEY,  -- Surrogate Key เป็น Primary Key หลัก
  EventCode      CHAR(3),                 -- Foreign Key เชื่อมไปยัง Event
  Score          NUMBER(4),      	  -- Score ของ Detail Event
  Discount       NUMBER(2),               -- Discount ของ Detail Event
  TicketCode     NUMBER(4),               -- Foreign Key เชื่อมไปยัง Ticket
  CONSTRAINT Fk_Event FOREIGN KEY (EventCode) REFERENCES Event(EventCode), 
  CONSTRAINT Fk_DetailEvent_Ticket FOREIGN KEY (TicketCode) REFERENCES Ticket(TicketCode)  
);
