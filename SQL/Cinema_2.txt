CREATE TABLE MemberType_B2 (
    MemTypeCode CHAR(3) PRIMARY KEY,
    MemTypeName VARCHAR2(20) NOT NULL,
    MemTypeDiscount NUMBER(2)
);

CREATE TABLE EventType_B2 (
    EvTypeCode CHAR(3) PRIMARY KEY,
    EvTypeName VARCHAR2(20) NOT NULL,
    EEvTypeDescription VARCHAR2(40)
);

CREATE TABLE Licensor_B2 (
    LicCode CHAR(3) PRIMARY KEY,
    LicName VARCHAR2(20) NOT NULL,
    LicEmail VARCHAR2(30) NOT NULL,
    LicTel VARCHAR2(15) NOT NULL
);

CREATE TABLE Member_B2 (
    MemCode CHAR(3) PRIMARY KEY,
    MemName VARCHAR2(20) NOT NULL,
    MemTel VARCHAR2(15) NOT NULL,
    MemSTDate DATE,
    MemEndDate DATE,
    MemTypeCode CHAR(3),
    CONSTRAINT Fk_MemType_B2 FOREIGN KEY (MemTypeCode)
        REFERENCES MemberType_B2(MemTypeCode)
);

CREATE TABLE Movie_B2 (
    MovieCode CHAR(3) PRIMARY KEY,
    MovieName VARCHAR2(20) NOT NULL,
    MovieTitle VARCHAR2(50) NOT NULL,
    MovieWriter VARCHAR2(20) NOT NULL,
    MovieDirector VARCHAR2(20) NOT NULL,
    MovieRating NUMBER(2),
    MovieDuration DATE,
    MovieCost NUMBER(4) NOT NULL,
    LicCode CHAR(3),
    CONSTRAINT Fk_Lic_B2 FOREIGN KEY (LicCode)
        REFERENCES Licensor_B2(LicCode)
);

CREATE TABLE Ticket_B2 (
    TicketCode NUMBER(4) PRIMARY KEY,
    TicketDate DATE,
    MemCode CHAR(3),
    CONSTRAINT Fk_Mem_B2 FOREIGN KEY (MemCode)
        REFERENCES Member_B2(MemCode)
);

CREATE TABLE ShowTime_B2 (
    SHTCode CHAR(4) PRIMARY KEY,
    SHTSTTime DATE,
    SHTEndTime DATE,
    SHTLang VARCHAR2(15),
    SHTCapacity NUMBER(3),
    MovieCode CHAR(3),
    CONSTRAINT Fk_Movie_B2 FOREIGN KEY (MovieCode)
        REFERENCES Movie_B2(MovieCode)
);

CREATE TABLE ShowTime_Ticket_B2 (
    SHTCode CHAR(4),
    TicketCode NUMBER(4),
    CONSTRAINT pk_ShowTime_Ticket_B2 PRIMARY KEY (SHTCode, TicketCode),
    CONSTRAINT fk_ShowTime_B2 FOREIGN KEY (SHTCode)
        REFERENCES ShowTime_B2(SHTCode),
    CONSTRAINT fk_Ticket_B2 FOREIGN KEY (TicketCode)
        REFERENCES Ticket_B2(TicketCode)
);

CREATE TABLE Event_B2 (
    EventCode CHAR(3) PRIMARY KEY,
    EventName VARCHAR2(20) NOT NULL,
    EventSTDate DATE,
    EvenEndDate DATE,
    EvTypeCode CHAR(3),
    SHTCode CHAR(4),
    CONSTRAINT Fk_EvType_B2 FOREIGN KEY (EvTypeCode)
        REFERENCES EventType_B2(EvTypeCode),
    CONSTRAINT Fk_SHT_B2 FOREIGN KEY (SHTCode)
        REFERENCES ShowTime_B2(SHTCode)
);

CREATE TABLE DetailEvent_B2 (
    DetailEventID NUMBER(10) PRIMARY KEY,
    EventCode CHAR(3),
    Score NUMBER(4),
    Discount NUMBER(2),
    TicketCode NUMBER(4),
    CONSTRAINT Fk_Event_B2 FOREIGN KEY (EventCode)
        REFERENCES Event_B2(EventCode),
    CONSTRAINT Fk_DetailEvent_Ticket_B2 FOREIGN KEY (TicketCode)
        REFERENCES Ticket_B2(TicketCode)
);
