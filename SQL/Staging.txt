CREATE TABLE ST_MemberType (
    MemTypeCode CHAR(3) PRIMARY KEY,
    MemTypeName VARCHAR2(20) NOT NULL,

    SourceSystem VARCHAR2(50),

CREATE TABLE ST_EventType (
    EvTypeCode CHAR(3) PRIMARY KEY,
    EvTypeName VARCHAR2(20) NOT NULL,

CREATE TABLE ST_Licensor (

    LicName VARCHAR2(20) NOT NULL,

    LicTel VARCHAR2(15) NOT NULL,
    SourceSystem VARCHAR2(50),

CREATE SEQUENCE ST_Member_Seq START WITH 1;
CREATE TABLE ST_Member (
    StagingID NUMBER PRIMARY KEY,
    MemCode CHAR(3),
    MemName VARCHAR2(20),
    MemTel VARCHAR2(15),
    MemSTDate DATE,
    MemEndDate DATE,
    MemTypeCode CHAR(3),
    SourceSystem VARCHAR2(50),
    InsertedTime TIMESTAMP DEFAULT SYSTIMESTAMP
);

CREATE OR REPLACE TRIGGER TRG_ST_Member
BEFORE INSERT ON ST_Member
FOR EACH ROW
BEGIN
  SELECT ST_Member_Seq.NEXTVAL INTO :NEW.StagingID FROM DUAL;
END;
/

MERGE INTO ST_MemberType tgt
USING (
  SELECT MemTypeCode, MemTypeName, MemTypeDiscount FROM cinema_1.MemberType
) src
ON (tgt.MemTypeCode = src.MemTypeCode)
WHEN MATCHED THEN
  UPDATE SET 
    tgt.MemTypeName = src.MemTypeName,
    tgt.MemTypeDiscount = src.MemTypeDiscount,
    tgt.SourceSystem = 'cinema_1',
    tgt.InsertedTime = SYSTIMESTAMP
  WHERE (tgt.MemTypeName != src.MemTypeName OR tgt.MemTypeDiscount != src.MemTypeDiscount)

WHEN NOT MATCHED THEN
  INSERT (MemTypeCode, MemTypeName, MemTypeDiscount, SourceSystem, InsertedTime)
  VALUES (src.MemTypeCode, src.MemTypeName, src.MemTypeDiscount, 'cinema_1', SYSTIMESTAMP);
